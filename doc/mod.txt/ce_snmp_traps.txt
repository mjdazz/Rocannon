NOTES                   *ce_snmp_traps-notes*

   Note

     • Recommended connection is network_cli.
     • This module also works with local connections for legacy playbooks.

EXAMPLES                   *ce_snmp_traps-examples*

 - name: CloudEngine snmp traps test
   hosts: cloudengine
   connection: local
   gather_facts: no
   vars:
     cli:
       host: "{{ inventory_hostname }}"
       port: "{{ ansible_ssh_port }}"
       username: "{{ username }}"
       password: "{{ password }}"
       transport: cli

   tasks:

   - name: "Config SNMP trap all enable"
     ce_snmp_traps:
       state: present
       feature_name: all
       provider: "{{ cli }}"

   - name: "Config SNMP trap interface"
     ce_snmp_traps:
       state: present
       interface_type: 40GE
       interface_number: 2/0/1
       provider: "{{ cli }}"

   - name: "Config SNMP trap port"
     ce_snmp_traps:
       state: present
       port_number: 2222
       provider: "{{ cli }}"
