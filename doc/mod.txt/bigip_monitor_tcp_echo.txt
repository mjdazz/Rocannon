EXAMPLES                   *bigip_monitor_tcp_echo-examples*

 - name: Create TCP Echo Monitor
   bigip_monitor_tcp_echo:
     state: present
     ip: 10.10.10.10
     name: my_tcp_monitor
     provider:
       password: secret
       server: lb.mydomain.com
       user: admin
   delegate_to: localhost

 - name: Remove TCP Echo Monitor
   bigip_monitor_tcp_echo:
     state: absent
     name: my_tcp_monitor
     provider:
       server: lb.mydomain.com
       user: admin
       password: secret
   delegate_to: localhost
