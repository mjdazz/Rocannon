EXAMPLES                   *ipa_sudocmdgroup-examples*

 - name: Ensure sudo command group exists
   ipa_sudocmdgroup:
     name: group01
     description: Group of important commands
     sudocmd:
     - su
     ipa_host: ipa.example.com
     ipa_user: admin
     ipa_pass: topsecret

 - name: Ensure sudo command group does not exist
   ipa_sudocmdgroup:
     name: group01
     state: absent
     ipa_host: ipa.example.com
     ipa_user: admin
     ipa_pass: topsecret
