EXAMPLES                   *aci_domain_to_encap_pool-examples*

 - name: Add domain to VLAN pool binding
   aci_domain_to_encap_pool:
     host: apic
     username: admin
     password: SomeSecretPassword
     domain: phys_dom
     domain_type: phys
     pool: test_pool
     pool_type: vlan
     pool_allocation_mode: dynamic
     state: present
   delegate_to: localhost

 - name: Remove domain to VLAN pool binding
   aci_domain_to_encap_pool:
     host: apic
     username: admin
     password: SomeSecretPassword
     domain: phys_dom
     domain_type: phys
     pool: test_pool
     pool_type: vlan
     pool_allocation_mode: dynamic
     state: absent
   delegate_to: localhost

 - name: Query our domain to VLAN pool binding
   aci_domain_to_encap_pool:
     host: apic
     username: admin
     password: SomeSecretPassword
     domain: phys_dom
     pool: test_pool
     pool_type: vlan
     pool_allocation_mode: dynamic
     state: query
   delegate_to: localhost
   register: query_result

 - name: Query all domain to VLAN pool bindings
   aci_domain_to_encap_pool:
     host: apic
     username: admin
     password: SomeSecretPassword
     domain_type: phys
     pool_type: vlan
     pool_allocation_mode: dynamic
     state: query
   delegate_to: localhost
   register: query_result
